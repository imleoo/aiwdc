Pipeline Configuration

pipeline.yaml
# AiWDC Pipeline Configuration
version: 2.0

sources:
  - name: rss-tech-news
    type: rss
    config:
      url: https://technews.example.com/rss
      poll_interval: 3600

  - name: web-scraper-news
    type: web
    config:
      urls: ["https://news.example.com", "https://blog.example.com"]
      selector: ".article-content"

  - name: database-articles
    type: database
    config:
      connection: ${DB_CONNECTION_STRING}
      query: "SELECT * FROM articles WHERE published = true"

  - name: search-engine-results
    type: search
    config:
      engine: google
      api_key: ${SEARCH_API_KEY}
      query: "latest technology trends"

ai_models:
  - name: aiwdc-processor
    type: aiwdc
    config:
      model_provider: dynamic
      default_model: gpt-4
      fallback_models: ["claude-3-opus", "qwen-max", "deepseek-v3"]
      temperature: 0.7
      max_tokens: 4000

destinations:
  - name: wordpress-blog
    type: wordpress
    config:
      url: https://myblog.com/wp-json
      username: ${WP_USERNAME}
      password: ${WP_PASSWORD}

  - name: shopify-store
    type: shopify
    config:
      store: mystore.myshopify.com
      access_token: ${SHOPIFY_TOKEN}

  - name: twitter-account
    type: twitter
    config:
      api_key: ${TWITTER_API_KEY}
      api_secret: ${TWITTER_API_SECRET}

  - name: facebook-page
    type: facebook
    config:
      page_id: ${FACEBOOK_PAGE_ID}
      access_token: ${FACEBOOK_ACCESS_TOKEN}

  - name: linkedin-company
    type: linkedin
    config:
      company_id: ${LINKEDIN_COMPANY_ID}
      access_token: ${LINKEDIN_ACCESS_TOKEN}

workflows:
  - name: tech-content-pipeline
    triggers:
      - schedule: "0 9 * * *"
    steps:
      - extract: [rss-tech-news, web-scraper-news, database-articles, search-engine-results]
      - transform: [openai-gpt4, claude-3, qwen-max, deepseek-v3]
      - load: [wordpress-blog, shopify-store, twitter-account, facebook-page, linkedin-company]
